import java.io.*;
import java.util.*;
public class Solution {
    public static void main(String[] args) {
        Scanner s = new Scanner(System.in);
        String name = s.next();
        int sub1 = s.nextInt();
        int sub2 = s.nextInt();
        int sub3 = s.nextInt();
        int sub4 = s.nextInt();
        int sub5 = s.nextInt();
        int totalmarks =(sub1+sub2+sub3+sub4+sub5);
        double average = (double)(sub1+sub2+sub3+sub4+sub5)/5;
        System.out.println("Name of the Student:"+name);
        System.out.println("Total Mark:"+totalmarks);
        System.out.printf("Average Mark:%.1f\n",average);
        if(average==100){
            System.out.println("Grade Mark:S");
        }else if(average >= 90 ){
            System.out.println("Grade Mark:A");
        }else if (average >=80 && average<= 89){
            System.out.println("Grade Mark:B");
        }else if(average >=70 && average <= 79){
            System.out.println("Grade Mark:C");
        }else if (average >=60 && average <= 69){
            System.out.println("Grade Mark:D");
        }else if (average >=50 && average <= 59){
            System.out.println("Grade Mark:E");
        }else{
            System.out.println("Grade Mark:Fail");
        }
    }
}
